13. 分页查询

分页实现

  • 在学习mysql数据库的时候我们都知道通过limit 关键字可以在数据库进行分页查询，例如 select * from tb_article limit 0,10; 但是在lucene中并未提供类似于 limit 的分页相关语法，开发者想要实现分页效果可以通过java程序来控制分页效果的实现；

  • 示例代码

    ``` import java.nio.file.Paths; import org.apache.lucene.analysis.Analyzer; import org.apache.lucene.analysis.cn.smart.SmartChineseAnalyzer; import org.apache.lucene.document.Document; import org.apache.lucene.index.DirectoryReader; import org.apache.lucene.index.IndexReader; import org.apache.lucene.queryparser.classic.QueryParser; import org.apache.lucene.search.IndexSearcher; import org.apache.lucene.search.Query; import org.apache.lucene.search.ScoreDoc; import org.apache.lucene.search.TopDocs; import org.apache.lucene.store.Directory; import org.apache.lucene.store.FSDirectory;

    public class IndexManager {

  public static void main(String[] args) {
      try {
          //指定索引库的目录
          Directory directory = FSDirectory.open(Paths.get("I:\\lucene_db\\tb_article"));

          //DirectoryReader的open方法指定需要读取的索引库信息，并返回相应的实例
          IndexReader indexReader = DirectoryReader.open(directory);

          //创建IndexSearcher对象，通过IndexSearcher对象进行全文检索
          IndexSearcher  indexSearcher = new IndexSearcher(indexReader);

          //创建智能分词器，对用户输入的信息进行分词   注意：索引的CRUD必须使用相同的分词器
          Analyzer analyzer = new SmartChineseAnalyzer();

          //创建QueryParse实例   通过QueryParse可以指定需要查询的字段 以及 分词器信息
          QueryParser queryParser = new QueryParser("title", analyzer);

          //通过QueryParser得到query实例,并指定用于查询的关键字
          Query query = queryParser.parse("长沙旅游景点");

          /*
           *查询目录区域，将匹配的记录封装在 TopDocs的对象中，search(query对象，最多返回多少条记录) 
           *lucene进行全文检索的时候会对满足条件的记录进行评分，评分越高说明匹配度越高
           *
           * */ 
          TopDocs topDocs = indexSearcher.search(query,10);

          //通过topDocs获取匹配的记录
          ScoreDoc[] scoreDocs = topDocs.scoreDocs;
          System.out.println("获取到的记录数："+scoreDocs.length);

          //获取查询到的记录数
          int totalNum = scoreDocs.length;
          /*****分页实现begin*****/
          //定义查询的页码
          int pageIndex = 1;

          //定义每页显示的记录数
          int pageSize = 4;

          //计算查询的起始行号 
          int startSize = (pageIndex - 1) * pageSize;
          //计算查询的结束行号
          int endSize = pageIndex * pageSize > totalNum?totalNum:pageIndex * pageSize;

          for (int i = startSize; i < endSize; i++) {
              //获取记录的id
              int id = scoreDocs[i].doc;
              //查询数据表
              Document document = indexSearcher.doc(id);
              String title = document.get("title");
              String content = document.get("content");
              System.out.println("title:"+title+" content:"+content);
          }
          /*****分页实现end*****/
      } catch (Exception e) {
          // TODO: handle exception
          e.printStackTrace();
      }
  }

} ```
